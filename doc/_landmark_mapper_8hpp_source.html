70  LandmarkMapper(boost::filesystem::path filename)
71  {
72  using std::string;
73  using boost::property_tree::ptree;
74  ptree configtree;
75  try {
76  boost::property_tree::info_parser::read_info(filename.string(), configtree);
77  }
78  catch (const boost::property_tree::ptree_error& error) {
79  throw std::runtime_error(string("LandmarkMapper: Error reading landmark-mappings file: ") + error.what());
80  }
81 
82  try {
83  ptree pt_landmark_mappings = configtree.get_child("landmarkMappings");
84  for (auto&& mapping : pt_landmark_mappings) {
85  landmark_mappings.insert(make_pair(mapping.first, mapping.second.get_value<string>()));
86  }
87  }
88  catch (const boost::property_tree::ptree_error& error) {
89  throw std::runtime_error(string("LandmarkMapper: Error while parsing the mappings file: ") + error.what());
90  }
91  catch (const std::runtime_error& error) {
92  throw std::runtime_error(string("LandmarkMapper: Error while parsing the mappings file: ") + error.what());
93  }

104  boost::optional<std::string> convert(std::string landmark_name) const
105  {
106  if (landmark_mappings.empty()) {
107  // perform identity mapping, i.e. return the input
108  return landmark_name;
109  }
110  else {
111  auto&& converted_landmark = landmark_mappings.find(landmark_name);
112  if (converted_landmark != std::end(landmark_mappings)) {
113  // landmark mapping found, return it
114  return converted_landmark->second;
115  }
116  else { // landmark_name does not match the key of any element in the map
117  return boost::none;
118  }
119  }
120  };
