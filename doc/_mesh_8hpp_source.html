103 inline void write_textured_obj(Mesh mesh, std::string filename)
104 {
105  assert((mesh.vertices.size() == mesh.colors.size() || mesh.colors.empty()) && !mesh.texcoords.empty());

111 
112  obj_file << "mtllib " << mtl_filename.string() << std::endl; // first line of the obj file
113 
114  // same as in write_obj():
115  if (mesh.colors.empty()) {
116  for (std::size_t i = 0; i < mesh.vertices.size(); ++i) {
117  obj_file << "v " << mesh.vertices[i][0] << " " << mesh.vertices[i][1] << " " << mesh.vertices[i][2] << " " << std::endl;
118  }
119  }
120  else {
121  for (std::size_t i = 0; i < mesh.vertices.size(); ++i) {
122  obj_file << "v " << mesh.vertices[i][0] << " " << mesh.vertices[i][1] << " " << mesh.vertices[i][2] << " " << mesh.colors[i][0] << " " << mesh.colors[i][1] << " " << mesh.colors[i][2] << " " << std::endl;
123  }
124  }
125  // end
126 
127  for (std::size_t i = 0; i < mesh.texcoords.size(); ++i) {
128  obj_file << "vt " << mesh.texcoords[i][0] << " " << 1.0f - mesh.texcoords[i][1] << std::endl;
129  // We invert y because Meshlab's uv origin (0, 0) is on the bottom-left
130  }
131 
132  obj_file << "usemtl FaceTexture" << std::endl; // the name of our texture (material) will be 'FaceTexture'
133 
134  for (auto&& v : mesh.tvi) {
135  // This assumes mesh.texcoords.size() == mesh.vertices.size(). The texture indices could theoretically be different (for example in the cube-mapped 3D scan)
136  // Add one because obj starts counting triangle indices at 1
137  obj_file << "f " << v[0] + 1 << "/" << v[0] + 1 << " " << v[1] + 1 << "/" << v[1] + 1 << " " << v[2] + 1 << "/" << v[2] + 1 << std::endl;
138  }
