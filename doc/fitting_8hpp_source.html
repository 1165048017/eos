68 inline Eigen::VectorXf fit_shape(cv::Mat affine_camera_matrix, const morphablemodel::MorphableModel& morphable_model, const std::vector<morphablemodel::Blendshape>& blendshapes, const std::vector<cv::Vec2f>& image_points, const std::vector<int>& vertex_indices, float lambda, boost::optional<int> num_coefficients_to_fit, std::vector<float>& pca_shape_coefficients, std::vector<float>& blendshape_coefficients)
69 {
70  using Eigen::VectorXf;
71  using Eigen::MatrixXf;
72 
73  MatrixXf blendshapes_as_basis = morphablemodel::to_matrix(blendshapes);
74 
75  std::vector<float> last_blendshape_coeffs, current_blendshape_coeffs;
76  std::vector<float> last_pca_coeffs, current_pca_coeffs;
77  current_blendshape_coeffs.resize(blendshapes.size()); // starting values t_0, all zeros
78  // no starting values for current_pca_coeffs required, since we start with the shape fitting, and cv::norm of an empty vector is 0.
79 
80  VectorXf combined_shape;
81  do // run at least once:
82  {
83  last_blendshape_coeffs = current_blendshape_coeffs;
84  last_pca_coeffs = current_pca_coeffs;
85  // Estimate the PCA shape coefficients with the current blendshape coefficients (0 in the first iteration):
86  VectorXf mean_plus_blendshapes = morphable_model.get_shape_model().get_mean() + blendshapes_as_basis * Eigen::Map<const Eigen::VectorXf>(last_blendshape_coeffs.data(), last_blendshape_coeffs.size());
87  current_pca_coeffs = fitting::fit_shape_to_landmarks_linear(morphable_model, affine_camera_matrix, image_points, vertex_indices, mean_plus_blendshapes, lambda, num_coefficients_to_fit);
88 
89  // Estimate the blendshape coefficients with the current PCA model estimate:
90  VectorXf pca_model_shape = morphable_model.get_shape_model().draw_sample(current_pca_coeffs);
91  current_blendshape_coeffs = fitting::fit_blendshapes_to_landmarks_nnls(blendshapes, pca_model_shape, affine_camera_matrix, image_points, vertex_indices);
92 
93  combined_shape = pca_model_shape + blendshapes_as_basis * Eigen::Map<const Eigen::VectorXf>(current_blendshape_coeffs.data(), current_blendshape_coeffs.size()); // Todo/Note: Could move outside the loop, not needed in here actually
94  } while (std::abs(cv::norm(current_pca_coeffs) - cv::norm(last_pca_coeffs)) >= 0.01 || std::abs(cv::norm(current_blendshape_coeffs) - cv::norm(last_blendshape_coeffs)) >= 0.01);
95 
96  pca_shape_coefficients = current_pca_coeffs;
97  blendshape_coefficients = current_blendshape_coeffs;
98 
99  return combined_shape;
100 };

144 inline auto get_corresponding_pointset(const T& landmarks, const core::LandmarkMapper& landmark_mapper, const morphablemodel::MorphableModel& morphable_model)
145 {
146  using cv::Mat;
147  using std::vector;
148  using cv::Vec2f;
149  using cv::Vec4f;
150 
151  // These will be the final 2D and 3D points used for the fitting:
152  vector<Vec4f> model_points; // the points in the 3D shape model
153  vector<int> vertex_indices; // their vertex indices
154  vector<Vec2f> image_points; // the corresponding 2D landmark points
155 
156  // Sub-select all the landmarks which we have a mapping for (i.e. that are defined in the 3DMM):
157  for (int i = 0; i < landmarks.size(); ++i) {
158  auto converted_name = landmark_mapper.convert(landmarks[i].name);
159  if (!converted_name) { // no mapping defined for the current landmark
160  continue;
161  }
162  int vertex_idx = std::stoi(converted_name.get());
163  auto vertex = morphable_model.get_shape_model().get_mean_at_point(vertex_idx);
164  model_points.emplace_back(Vec4f(vertex.x(), vertex.y(), vertex.z(), 1.0f));
165  vertex_indices.emplace_back(vertex_idx);
166  image_points.emplace_back(landmarks[i].coordinates);
167  }
168  return std::make_tuple(image_points, model_points, vertex_indices);
169 };

232 inline std::pair<core::Mesh, fitting::RenderingParameters> fit_shape_and_pose(const morphablemodel::MorphableModel& morphable_model, const std::vector<morphablemodel::Blendshape>& blendshapes, const core::LandmarkCollection<cv::Vec2f>& landmarks, const core::LandmarkMapper& landmark_mapper, int image_width, int image_height, const morphablemodel::EdgeTopology& edge_topology, const fitting::ContourLandmarks& contour_landmarks, const fitting::ModelContour& model_contour, int num_iterations, boost::optional<int> num_shape_coefficients_to_fit, float lambda, boost::optional<fitting::RenderingParameters> initial_rendering_params, std::vector<float>& pca_shape_coefficients, std::vector<float>& blendshape_coefficients, std::vector<cv::Vec2f>& fitted_image_points)
233 {
234  assert(blendshapes.size() > 0);
235  assert(landmarks.size() >= 4);
236  assert(image_width > 0 && image_height > 0);
237  assert(num_iterations > 0); // Can we allow 0, for only the initial pose-fit?
238  assert(pca_shape_coefficients.size() <= morphable_model.get_shape_model().get_num_principal_components());
239  // More asserts I forgot?
240 
241  using std::vector;
242  using cv::Vec2f;
243  using cv::Vec4f;
244  using Eigen::VectorXf;
245  using Eigen::MatrixXf;
246 
247  if (!num_shape_coefficients_to_fit)
248  {
249  num_shape_coefficients_to_fit = morphable_model.get_shape_model().get_num_principal_components();
250  }
251 
252  if (pca_shape_coefficients.empty())
253  {
254  pca_shape_coefficients.resize(num_shape_coefficients_to_fit.get());
255  }
256  // Todo: This leaves the following case open: num_coeffs given is empty or defined, but the
257  // pca_shape_coefficients given is != num_coeffs or the model's max-coeffs. What to do then? Handle & document!
258 
259  if (blendshape_coefficients.empty())
260  {
261  blendshape_coefficients.resize(blendshapes.size());
262  }
263 
264  MatrixXf blendshapes_as_basis = morphablemodel::to_matrix(blendshapes);
265 
266  // Current mesh - either from the given coefficients, or the mean:
267  VectorXf current_pca_shape = morphable_model.get_shape_model().draw_sample(pca_shape_coefficients);
268  VectorXf current_combined_shape = current_pca_shape + blendshapes_as_basis * Eigen::Map<const Eigen::VectorXf>(blendshape_coefficients.data(), blendshape_coefficients.size());
269  auto current_mesh = morphablemodel::sample_to_mesh(current_combined_shape, morphable_model.get_color_model().get_mean(), morphable_model.get_shape_model().get_triangle_list(), morphable_model.get_color_model().get_triangle_list(), morphable_model.get_texture_coordinates());
270 
271  // The 2D and 3D point correspondences used for the fitting:
272  vector<Vec4f> model_points; // the points in the 3D shape model
273  vector<int> vertex_indices; // their vertex indices
274  vector<Vec2f> image_points; // the corresponding 2D landmark points
275 
276  // Sub-select all the landmarks which we have a mapping for (i.e. that are defined in the 3DMM),
277  // and get the corresponding model points (mean if given no initial coeffs, from the computed shape otherwise):
278  for (int i = 0; i < landmarks.size(); ++i) {
279  auto converted_name = landmark_mapper.convert(landmarks[i].name);
280  if (!converted_name) { // no mapping defined for the current landmark
281  continue;
282  }
283  int vertex_idx = std::stoi(converted_name.get());
284  Vec4f vertex(current_mesh.vertices[vertex_idx].x, current_mesh.vertices[vertex_idx].y, current_mesh.vertices[vertex_idx].z, current_mesh.vertices[vertex_idx].w);
285  model_points.emplace_back(vertex);
286  vertex_indices.emplace_back(vertex_idx);
287  image_points.emplace_back(landmarks[i].coordinates);
288  }
289 
290  // Need to do an initial pose fit to do the contour fitting inside the loop.
291  // We'll do an expression fit too, since face shapes vary quite a lot, depending on expressions.
292  fitting::ScaledOrthoProjectionParameters current_pose;
293  current_pose = fitting::estimate_orthographic_projection_linear(image_points, model_points, true, image_height);
294  fitting::RenderingParameters rendering_params(current_pose, image_width, image_height);
295 
296  cv::Mat affine_from_ortho = fitting::get_3x4_affine_camera_matrix(rendering_params, image_width, image_height);
297  blendshape_coefficients = fitting::fit_blendshapes_to_landmarks_nnls(blendshapes, current_pca_shape, affine_from_ortho, image_points, vertex_indices);
298 
299  // Mesh with same PCA coeffs as before, but new expression fit (this is relevant if no initial blendshape coeffs have been given):
300  current_combined_shape = current_pca_shape + morphablemodel::to_matrix(blendshapes) * Eigen::Map<const Eigen::VectorXf>(blendshape_coefficients.data(), blendshape_coefficients.size());
301  current_mesh = morphablemodel::sample_to_mesh(current_combined_shape, morphable_model.get_color_model().get_mean(), morphable_model.get_shape_model().get_triangle_list(), morphable_model.get_color_model().get_triangle_list(), morphable_model.get_texture_coordinates());
302 
303  // The static (fixed) landmark correspondences which will stay the same throughout
304  // the fitting (the inner face landmarks):
305  auto fixed_image_points = image_points;
306  auto fixed_vertex_indices = vertex_indices;
307 
308  for (int i = 0; i < num_iterations; ++i)
309  {
310  image_points = fixed_image_points;
311  vertex_indices = fixed_vertex_indices;
312  // Given the current pose, find 2D-3D contour correspondences of the front-facing face contour:
313  vector<Vec2f> image_points_contour;
314  vector<int> vertex_indices_contour;
315  auto yaw_angle = glm::degrees(glm::eulerAngles(rendering_params.get_rotation())[1]);
316  // For each 2D contour landmark, get the corresponding 3D vertex point and vertex id:
317  std::tie(image_points_contour, std::ignore, vertex_indices_contour) = fitting::get_contour_correspondences(landmarks, contour_landmarks, model_contour, yaw_angle, current_mesh, rendering_params.get_modelview(), rendering_params.get_projection(), fitting::get_opencv_viewport(image_width, image_height));
318  // Add the contour correspondences to the set of landmarks that we use for the fitting:
319  vertex_indices = fitting::concat(vertex_indices, vertex_indices_contour);
320  image_points = fitting::concat(image_points, image_points_contour);
321 
322  // Fit the occluding (away-facing) contour using the detected contour LMs:
323  vector<Eigen::Vector2f> occluding_contour_landmarks;
324  if (yaw_angle >= 0.0f) // positive yaw = subject looking to the left
325  { // the left contour is the occluding one we want to use ("away-facing")
326  auto contour_landmarks_ = core::filter(landmarks, contour_landmarks.left_contour); // Can do this outside of the loop
327  std::for_each(begin(contour_landmarks_), end(contour_landmarks_), [&occluding_contour_landmarks](auto&& lm) { occluding_contour_landmarks.push_back({ lm.coordinates[0], lm.coordinates[1] }); });
328  }
329  else {
330  auto contour_landmarks_ = core::filter(landmarks, contour_landmarks.right_contour);
331  std::for_each(begin(contour_landmarks_), end(contour_landmarks_), [&occluding_contour_landmarks](auto&& lm) { occluding_contour_landmarks.push_back({ lm.coordinates[0], lm.coordinates[1] }); });
332  }
333  auto edge_correspondences = fitting::find_occluding_edge_correspondences(current_mesh, edge_topology, rendering_params, occluding_contour_landmarks, 180.0f);
334  image_points = fitting::concat(image_points, edge_correspondences.first);
335  vertex_indices = fitting::concat(vertex_indices, edge_correspondences.second);
336 
337  // Get the model points of the current mesh, for all correspondences that we've got:
338  model_points.clear();
339  for (const auto& v : vertex_indices)
340  {
341  model_points.push_back({ current_mesh.vertices[v][0], current_mesh.vertices[v][1], current_mesh.vertices[v][2], current_mesh.vertices[v][3] });
342  }
343 
344  // Re-estimate the pose, using all correspondences:
345  current_pose = fitting::estimate_orthographic_projection_linear(image_points, model_points, true, image_height);
346  rendering_params = fitting::RenderingParameters(current_pose, image_width, image_height);
347 
348  cv::Mat affine_from_ortho = fitting::get_3x4_affine_camera_matrix(rendering_params, image_width, image_height);
349 
350  // Estimate the PCA shape coefficients with the current blendshape coefficients:
351  VectorXf mean_plus_blendshapes = morphable_model.get_shape_model().get_mean() + blendshapes_as_basis * Eigen::Map<const Eigen::VectorXf>(blendshape_coefficients.data(), blendshape_coefficients.size());
352  pca_shape_coefficients = fitting::fit_shape_to_landmarks_linear(morphable_model, affine_from_ortho, image_points, vertex_indices, mean_plus_blendshapes, lambda, num_shape_coefficients_to_fit);
353 
354  // Estimate the blendshape coefficients with the current PCA model estimate:
355  current_pca_shape = morphable_model.get_shape_model().draw_sample(pca_shape_coefficients);
356  blendshape_coefficients = fitting::fit_blendshapes_to_landmarks_nnls(blendshapes, current_pca_shape, affine_from_ortho, image_points, vertex_indices);
357 
358  current_combined_shape = current_pca_shape + blendshapes_as_basis * Eigen::Map<const Eigen::VectorXf>(blendshape_coefficients.data(), blendshape_coefficients.size());
359  current_mesh = morphablemodel::sample_to_mesh(current_combined_shape, morphable_model.get_color_model().get_mean(), morphable_model.get_shape_model().get_triangle_list(), morphable_model.get_color_model().get_triangle_list(), morphable_model.get_texture_coordinates());
360  }
361 
362  fitted_image_points = image_points;
363  return { current_mesh, rendering_params }; // I think we could also work with a Mat face_instance in this function instead of a Mesh, but it would convolute the code more (i.e. more complicated to access vertices).
364 };
