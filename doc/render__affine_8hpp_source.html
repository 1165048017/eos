56 inline std::pair<cv::Mat, cv::Mat> render_affine(const core::Mesh& mesh, cv::Mat affine_camera_matrix, int viewport_width, int viewport_height, bool do_backface_culling = true)
57 {
58  assert(mesh.vertices.size() == mesh.colors.size() || mesh.colors.empty()); // The number of vertices has to be equal for both shape and colour, or, alternatively, it has to be a shape-only model.
59  //assert(mesh.vertices.size() == mesh.texcoords.size() || mesh.texcoords.empty()); // same for the texcoords
60 
61  using cv::Mat;
62  using std::vector;
63 
64  Mat colourbuffer = Mat::zeros(viewport_height, viewport_width, CV_8UC4);
65  Mat depthbuffer = std::numeric_limits<float>::max() * Mat::ones(viewport_height, viewport_width, CV_64FC1);
66 
67  Mat affine_with_z = detail::calculate_affine_z_direction(affine_camera_matrix);
68 
69  vector<detail::Vertex<float>> projected_vertices;
70  projected_vertices.reserve(mesh.vertices.size());
71  for (int i = 0; i < mesh.vertices.size(); ++i) {
72  Mat vertex_screen_coords = affine_with_z * Mat(cv::Vec4f(mesh.vertices[i].x, mesh.vertices[i].y, mesh.vertices[i].z, mesh.vertices[i].w));
73  glm::tvec4<float> vertex_screen_coords_glm(vertex_screen_coords.at<float>(0), vertex_screen_coords.at<float>(1), vertex_screen_coords.at<float>(2), vertex_screen_coords.at<float>(3));
74  glm::tvec3<float> vertex_colour;
75  if (mesh.colors.empty()) {
76  vertex_colour = glm::tvec3<float>(0.5f, 0.5f, 0.5f);
77  }
78  else {
79  vertex_colour = mesh.colors[i];
80  }
81  projected_vertices.push_back(detail::Vertex<float>{vertex_screen_coords_glm, vertex_colour, mesh.texcoords[i]});
82  }
83 
84  // All vertices are screen-coordinates now
85  vector<detail::TriangleToRasterize> triangles_to_raster;
86  for (const auto& tri_indices : mesh.tvi) {
87  if (do_backface_culling) {
88  if (!detail::are_vertices_ccw_in_screen_space(glm::tvec2<float>(projected_vertices[tri_indices[0]].position), glm::tvec2<float>(projected_vertices[tri_indices[1]].position), glm::tvec2<float>(projected_vertices[tri_indices[2]].position)))
89  continue; // don't render this triangle
90  }
91 
92  // Get the bounding box of the triangle:
93  // take care: What do we do if all 3 vertices are not visible. Seems to work on a test case.
94  cv::Rect bounding_box = detail::calculate_clipped_bounding_box(glm::tvec2<float>(projected_vertices[tri_indices[0]].position), glm::tvec2<float>(projected_vertices[tri_indices[1]].position), glm::tvec2<float>(projected_vertices[tri_indices[2]].position), viewport_width, viewport_height);
95  auto min_x = bounding_box.x;
96  auto max_x = bounding_box.x + bounding_box.width;
97  auto min_y = bounding_box.y;
98  auto max_y = bounding_box.y + bounding_box.height;
99 
100  if (max_x <= min_x || max_y <= min_y) // Note: Can the width/height of the bbox be negative? Maybe we only need to check for equality here?
101  continue;
102 
103  detail::TriangleToRasterize t;
104  t.min_x = min_x;
105  t.max_x = max_x;
106  t.min_y = min_y;
107  t.max_y = max_y;
108  t.v0 = projected_vertices[tri_indices[0]];
109  t.v1 = projected_vertices[tri_indices[1]];
110  t.v2 = projected_vertices[tri_indices[2]];
111 
112  triangles_to_raster.push_back(t);
113  }
114 
115  // Raster all triangles, i.e. colour the pixel values and write the z-buffer
116  for (auto&& triangle : triangles_to_raster) {
117  detail::raster_triangle_affine(triangle, colourbuffer, depthbuffer);
118  }
119  return std::make_pair(colourbuffer, depthbuffer);
120 };
