70 inline ScaledOrthoProjectionParameters estimate_orthographic_projection_linear(std::vector<cv::Vec2f> image_points, std::vector<cv::Vec4f> model_points, bool is_viewport_upsidedown, boost::optional<int> viewport_height = boost::none)
71 {
72  using cv::Mat;
73  assert(image_points.size() == model_points.size());
74  assert(image_points.size() >= 4); // Number of correspondence points given needs to be equal to or larger than 4
75 
76  const int num_correspondences = static_cast<int>(image_points.size());
77 
78  if (is_viewport_upsidedown)
79  {
80  if (viewport_height == boost::none)
81  {
82  throw std::runtime_error("Error: If is_viewport_upsidedown is set to true, viewport_height needs to be given.");
83  }
84  for (auto&& ip : image_points)
85  {
86  ip[1] = viewport_height.get() - ip[1];
87  }
88  }
89 
90  Mat A = Mat::zeros(2 * num_correspondences, 8, CV_32FC1);
91  int row_index = 0;
92  for (int i = 0; i < model_points.size(); ++i)
93  {
94  Mat p = Mat(model_points[i]).t();
95  p.copyTo(A.row(row_index++).colRange(0, 4)); // even row - copy to left side (first row is row 0)
96  p.copyTo(A.row(row_index++).colRange(4, 8)); // odd row - copy to right side
97  } // 4th coord (homogeneous) is already 1
98 
99  Mat b(2 * num_correspondences, 1, CV_32FC1);
100  row_index = 0;
101  for (int i = 0; i < image_points.size(); ++i)
102  {
103  b.at<float>(row_index++) = image_points[i][0];
104  b.at<float>(row_index++) = image_points[i][1];
105  }
106 
107  Mat k; // resulting affine matrix (8x1)
108  bool solved = cv::solve(A, b, k, cv::DECOMP_SVD);
109 
110  const Mat R1 = k.rowRange(0, 3);
111  const Mat R2 = k.rowRange(4, 7);
112  const float sTx = k.at<float>(3);
113  const float sTy = k.at<float>(7);
114  const auto s = (cv::norm(R1) + cv::norm(R2)) / 2.0;
115  Mat r1 = R1 / cv::norm(R1);
116  Mat r2 = R2 / cv::norm(R2);
117  Mat r3 = r1.cross(r2);
118  Mat R;
119  r1 = r1.t();
120  r2 = r2.t();
121  r3 = r3.t();
122  R.push_back(r1);
123  R.push_back(r2);
124  R.push_back(r3);
125  // Set R to the closest orthonormal matrix to the estimated affine transform:
126  Mat S, U, Vt;
127  cv::SVDecomp(R, S, U, Vt);
128  Mat R_ortho = U * Vt;
129  // The determinant of R must be 1 for it to be a valid rotation matrix
130  if (cv::determinant(R_ortho) < 0)
131  {
132  U.row(2) = -U.row(2); // not sure this works...
133  R_ortho = U * Vt;
134  }
135 
136  // Remove the scale from the translations:
137  const auto t1 = sTx / s;
138  const auto t2 = sTy / s;
139 
140  // Convert to a glm::mat4x4:
141  glm::mat3x3 R_glm; // identity
142  for (int r = 0; r < 3; ++r) {
143  for (int c = 0; c < 3; ++c) {
144  R_glm[c][r] = R_ortho.at<float>(r, c);
145  }
146  }
147  return ScaledOrthoProjectionParameters{ R_glm, t1, t2, s };
148 };
