284 inline cv::Mat normalise_pca_basis(cv::Mat unnormalised_basis, cv::Mat eigenvalues)
285 {
286  using cv::Mat;
287  Mat normalised_basis(unnormalised_basis.size(), unnormalised_basis.type()); // empty matrix with the same dimensions
288  Mat sqrt_of_eigenvalues = eigenvalues.clone();
289  for (int i = 0; i < eigenvalues.rows; ++i) {
290  sqrt_of_eigenvalues.at<float>(i) = std::sqrt(eigenvalues.at<float>(i));
291  }
292  // Normalise the basis: We multiply each eigenvector (i.e. each column) with the square root of its corresponding eigenvalue
293  for (int basis = 0; basis < unnormalised_basis.cols; ++basis) {
294  Mat normalised_eigenvector = unnormalised_basis.col(basis).mul(sqrt_of_eigenvalues.at<float>(basis));
295  normalised_eigenvector.copyTo(normalised_basis.col(basis));
296  }
297 
298  return normalised_basis;
299 };

311 inline cv::Mat unnormalise_pca_basis(cv::Mat normalised_basis, cv::Mat eigenvalues)
312 {
313  using cv::Mat;
314  Mat unnormalised_basis(normalised_basis.size(), normalised_basis.type()); // empty matrix with the same dimensions
315  Mat one_over_sqrt_of_eigenvalues = eigenvalues.clone();
316  for (int i = 0; i < eigenvalues.rows; ++i) {
317  one_over_sqrt_of_eigenvalues.at<float>(i) = 1.0f / std::sqrt(eigenvalues.at<float>(i));
318  }
319  // De-normalise the basis: We multiply each eigenvector (i.e. each column) with 1 over the square root of its corresponding eigenvalue
320  for (int basis = 0; basis < normalised_basis.cols; ++basis) {
321  Mat unnormalised_eigenvector = normalised_basis.col(basis).mul(one_over_sqrt_of_eigenvalues.at<float>(basis));
322  unnormalised_eigenvector.copyTo(unnormalised_basis.col(basis));
323  }
324 
325  return unnormalised_basis;
326 };
