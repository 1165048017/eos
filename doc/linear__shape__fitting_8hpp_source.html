60 inline std::vector<float> fit_shape_to_landmarks_linear(const morphablemodel::MorphableModel& morphable_model, cv::Mat affine_camera_matrix, const std::vector<cv::Vec2f>& landmarks, const std::vector<int>& vertex_ids, Eigen::VectorXf base_face=Eigen::VectorXf(), float lambda=3.0f, boost::optional<int> num_coefficients_to_fit=boost::optional<int>(), boost::optional<float> detector_standard_deviation=boost::optional<float>(), boost::optional<float> model_standard_deviation=boost::optional<float>())
61 {
62  assert(landmarks.size() == vertex_ids.size());
63 
64  using Eigen::VectorXf;
65  using Eigen::MatrixXf;
66 
67  int num_coeffs_to_fit = num_coefficients_to_fit.get_value_or(morphable_model.get_shape_model().get_num_principal_components());
68  int num_landmarks = static_cast<int>(landmarks.size());
69 
70  if (base_face.size() == 0)
71  {
72  base_face = morphable_model.get_shape_model().get_mean();
73  }
74 
75  // $\hat{V} \in R^{3N\times m-1}$, subselect the rows of the eigenvector matrix $V$ associated with the $N$ feature points
76  // And we insert a row of zeros after every third row, resulting in matrix $\hat{V}_h \in R^{4N\times m-1}$:
77  MatrixXf V_hat_h = MatrixXf::Zero(4 * num_landmarks, num_coeffs_to_fit);
78  int row_index = 0;
79  for (int i = 0; i < num_landmarks; ++i) {
80  MatrixXf basis_rows_ = morphable_model.get_shape_model().get_rescaled_pca_basis_at_point(vertex_ids[i]); // In the paper, the orthonormal basis might be used? I'm not sure, check it. It's even a mess in the paper. PH 26.5.2014: I think the rescaled basis is fine/better.
81  V_hat_h.block(row_index, 0, 3, V_hat_h.cols()) = basis_rows_.block(0, 0, basis_rows_.rows(), num_coeffs_to_fit);
82  row_index += 4; // replace 3 rows and skip the 4th one, it has all zeros
83  }
84  // Form a block diagonal matrix $P \in R^{3N\times 4N}$ in which the camera matrix C (P_Affine, affine_camera_matrix) is placed on the diagonal:
85  MatrixXf P = MatrixXf::Zero(3 * num_landmarks, 4 * num_landmarks);
86  for (int i = 0; i < num_landmarks; ++i) {
87  using RowMajorMatrixXf = Eigen::Matrix<float, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor>;
88  P.block(3 * i, 4 * i, 3, 4) = Eigen::Map<RowMajorMatrixXf>(affine_camera_matrix.ptr<float>(), affine_camera_matrix.rows, affine_camera_matrix.cols);
89  }
90  // The variances: Add the 2D and 3D standard deviations.
91  // If the user doesn't provide them, we choose the following:
92  // 2D (detector) standard deviation: In pixel, we follow [1] and choose sqrt(3) as the default value.
93  // 3D (model) variance: 0.0f. It only makes sense to set it to something when we have a different variance for different vertices.
94  // The 3D variance has to be projected to 2D (for details, see paper [1]) so the units do match up.
95  float sigma_squared_2D = std::pow(detector_standard_deviation.get_value_or(std::sqrt(3.0f)), 2) + std::pow(model_standard_deviation.get_value_or(0.0f), 2);
96  // We use a VectorXf, and later use .asDiagonal():
97  VectorXf Omega = VectorXf::Constant(3 * num_landmarks, 1.0f / sigma_squared_2D);
98  // Earlier, we set Sigma in a for-loop and then computed Omega, but it was really unnecessary:
99  // Sigma(i, i) = sqrt(sigma_squared_2D), but then Omega is Sigma.t() * Sigma (squares the diagonal) - so we just assign 1/sigma_squared_2D to Omega here.
100 
101  // The landmarks in matrix notation (in homogeneous coordinates), $3N\times 1$
102  VectorXf y = VectorXf::Ones(3 * num_landmarks);
103  for (int i = 0; i < num_landmarks; ++i) {
104  y(3 * i) = landmarks[i][0];
105  y((3 * i) + 1) = landmarks[i][1];
106  //y((3 * i) + 2) = 1; // already 1, stays (homogeneous coordinate)
107  }
108  // The mean, with an added homogeneous coordinate (x_1, y_1, z_1, 1, x_2, ...)^t
109  VectorXf v_bar = VectorXf::Ones(4 * num_landmarks);
110  for (int i = 0; i < num_landmarks; ++i) {
111  v_bar(4 * i) = base_face(vertex_ids[i] * 3);
112  v_bar((4 * i) + 1) = base_face(vertex_ids[i] * 3 + 1);
113  v_bar((4 * i) + 2) = base_face(vertex_ids[i] * 3 + 2);
114  //v_bar((4 * i) + 3) = 1; // already 1, stays (homogeneous coordinate)
115  }
116 
117  // Bring into standard regularised quadratic form with diagonal distance matrix Omega:
118  const MatrixXf A = P * V_hat_h; // camera matrix times the basis
119  const MatrixXf b = P * v_bar - y; // camera matrix times the mean, minus the landmarks
120  const MatrixXf AtOmegaAReg = A.transpose() * Omega.asDiagonal() * A + lambda * Eigen::MatrixXf::Identity(num_coeffs_to_fit, num_coeffs_to_fit);
121  const MatrixXf rhs = -A.transpose() * Omega.asDiagonal() * b; // It's -A^t*Omega^t*b, but we don't need to transpose Omega, since it's a diagonal matrix, and Omega^t = Omega.
122  // c_s: The 'x' that we solve for. (The variance-normalised shape parameter vector, $c_s = [a_1/sigma_{s,1} , ..., a_m-1/sigma_{s,m-1}]^t$.)
123  // We get coefficients ~ N(0, 1), because we're fitting with the rescaled basis. The coefficients are not multiplied with their eigenvalues.
124  const VectorXf c_s = AtOmegaAReg.colPivHouseholderQr().solve(rhs);
125 
126  return std::vector<float>(c_s.data(), c_s.data() + c_s.size());
127 };
