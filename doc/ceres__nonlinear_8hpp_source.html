125  template <typename T>
126  bool operator()(const T* const camera_rotation, const T* const camera_translation_and_intrinsics, const T* const shape_coeffs, const T* const blendshape_coeffs, T* residual) const
127  {
128  using namespace glm;
129  // Generate shape instance (of only one vertex id!) using current parameters and 10 shape coefficients:
130  // Note: Why are we not returning a glm::tvec3<T>?
131  const auto point_arr = get_shape_point<T>(shape_model, blendshapes, vertex_id, shape_coeffs, blendshape_coeffs);
132 
133  // Project the point to 2D:
134  const tvec3<T> point_3d(point_arr[0], point_arr[1], point_arr[2]);
135  // I think the quaternion is always normalised because we run Ceres with QuaternionParameterization
136  const tquat<T> rot_quat(camera_rotation[0], camera_rotation[1], camera_rotation[2], camera_rotation[3]);
137  // We rotate ZXY*p, which is RPY*p. I'm not sure this matrix still corresponds to RPY - probably if we use glm::eulerAngles(), these are not RPY anymore and we'd have to adjust if we were to use rotation matrices.
138  const tmat4x4<T> rot_mtx = glm::mat4_cast(rot_quat);
139 
140  // Todo: use get_opencv_viewport() from nonlin_cam_esti.hpp.
141  const tvec4<T> viewport(0, image_height, image_width, -image_height); // OpenCV convention
142 
143  tvec3<T> projected_point; // Note: could avoid default construction by using a lambda and immediate invocation
144  if (use_perspective)
145  {
146  const auto t_mtx = glm::translate(tvec3<T>(camera_translation_and_intrinsics[0], camera_translation_and_intrinsics[1], camera_translation_and_intrinsics[2]));
147  const T& fov = camera_translation_and_intrinsics[3];
148  const auto persp_mtx = glm::perspective(fov, T(aspect_ratio), T(0.1), T(1000.0));
149  projected_point = glm::project(point_3d, t_mtx * rot_mtx, persp_mtx, viewport);
150  }
151  else {
152  const T& frustum_scale = camera_translation_and_intrinsics[2];
153  const auto t_mtx = glm::translate(tvec3<T>(camera_translation_and_intrinsics[0], camera_translation_and_intrinsics[1], 0.0)); // we don't have t_z in ortho camera, it doesn't matter where it is
154  const auto ortho_mtx = glm::ortho(-1.0 * aspect_ratio * frustum_scale, 1.0 * aspect_ratio * frustum_scale, -1.0 * frustum_scale, 1.0 * frustum_scale);
155  projected_point = glm::project(point_3d, t_mtx * rot_mtx, ortho_mtx, viewport);
156  }
157  // Residual: Projected point minus the observed 2D landmark point
158  residual[0] = projected_point.x - T(observed_landmark[0]);
159  residual[1] = projected_point.y - T(observed_landmark[1]);
160  return true;
161  };

316 template <typename T>
317 std::array<T, 3> get_shape_point(const morphablemodel::PcaModel& shape_model, const std::vector<morphablemodel::Blendshape>& blendshapes, int vertex_id, const T* const shape_coeffs, const T* const blendshape_coeffs)
318 {
319  int num_coeffs_fitting = 10; // Todo: Should be inferred or a function parameter!
320  auto mean = shape_model.get_mean_at_point(vertex_id);
321  auto basis = shape_model.get_rescaled_pca_basis_at_point(vertex_id);
322  // Computing Shape = mean + basis * coeffs:
323  // Note: Could use an Eigen matrix with type T to see if it gives a speedup.
324  std::array<T, 3> point{ T(mean[0]), T(mean[1]), T(mean[2]) };
325  for (int i = 0; i < num_coeffs_fitting; ++i) {
326  point[0] += T(basis.row(0).col(i)(0)) * shape_coeffs[i]; // it seems to be ~15% faster when these are static_cast<double>() instead of T()?
327  }
328  for (int i = 0; i < num_coeffs_fitting; ++i) {
329  point[1] += T(basis.row(1).col(i)(0)) * shape_coeffs[i];
330  }
331  for (int i = 0; i < num_coeffs_fitting; ++i) {
332  point[2] += T(basis.row(2).col(i)(0)) * shape_coeffs[i];
333  }
334  // Adding the blendshape offsets:
335  // Shape = mean + basis * coeffs + blendshapes * bs_coeffs:
336  auto num_blendshapes = blendshapes.size();
337  for (int i = 0; i < num_blendshapes; ++i) {
338  point[0] += T(blendshapes[i].deformation(3 * vertex_id + 0)) * blendshape_coeffs[i];
339  }
340  for (int i = 0; i < num_blendshapes; ++i) {
341  point[1] += T(blendshapes[i].deformation(3 * vertex_id + 1)) * blendshape_coeffs[i];
342  }
343  for (int i = 0; i < num_blendshapes; ++i) {
344  point[2] += T(blendshapes[i].deformation(3 * vertex_id + 2)) * blendshape_coeffs[i];
345  }
346  return point;
347 };
